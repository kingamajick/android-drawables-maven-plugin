31  /**
32   * 
33   * Copies all files with the extension (<code>'.png'</code>, <code>'.jpg'</code>, <code>'.gif'</code>) from any defined resource directories
34   * to <code>'${project.build.outputDirectory}/res'</code>. Only files contained in the following directories will be considered:
35   * <ul>
36   * <li>ldpi</li>
37   * <li>mdpi</li>
38   * <li>hdpi</li>
39   * <li>xhdpi</li>
40   * <li>nodpi</li>
41   * <li>tvdpi</li>
42   * </ul>
43   * Any files contained within sub directories of the above will have their name created from their path. For example
44   * <code>/${subDir1}/${subDir2}/${imageFileName}</code> will be named <code>${subDir1}_${subDir2}_${imageFileName}</code>. All names will be
45   * in lower case.
46   * 
47   * @author R King
48   * 
49   * @goal static-resources
50   */
51  // TODO: Check for bad drawable names
52  public class ProcessBitmapResourcesMojo extends AbstractMojo {
53  
54  	/**
55  	 * @parameter expression="${project.resources}"
56  	 * @readonly
57  	 */
58  	List<Resource> resources;
59  
60  	/**
61  	 * @parameter expression="${project.build.outputDirectory}/res"
62  	 * @readonly
63  	 */
64  	File targetDir;
65  
66  	/*
67  	 * (non-Javadoc)
68  	 * 
69  	 * @see org.apache.maven.plugin.Mojo#execute()
70  	 */
71  	public void execute() throws MojoExecutionException, MojoFailureException {
72  		if (this.resources.size() == 0) {
73  			return;
74  		}
75  		for (Resource resource : this.resources) {
76  			File resourcesDir = new File(resource.getDirectory());
77  			for (String imageResolutionDirName : Constants.STATIC_IMAGE_DIR) {
78  				File imageResolutionDir = new File(resourcesDir, imageResolutionDirName);
79  				if (imageResolutionDir.exists() || imageResolutionDir.isDirectory()) {
80  					try {
81  						copyImageResources(new File(this.targetDir, Constants.DRAWABLE_PREFIX + imageResolutionDirName), imageResolutionDir);
82  					}
83  					catch (IOException e) {
84  						throw new MojoFailureException("Unable to copy static resources", e);
85  					}
86  				}
87  				else {
88  					getLog().info("No image resources defined for " + imageResolutionDirName);
89  					continue;
90  				}
91  			}
92  		}
93  	}
94  
95  	/**
96  	 * Copies all images resources ({@link Constants#IMAGE_TYPES}) found in the list of files (and its sub directories) to the output
97  	 * directory. All images will be given lower case names and any images contained in sub directories will be named as follows:
98  	 * ${subDir1}_${subDir2}_${imageFileName}, these steps guarantee compatibility with android resource file names restrictions.
99  	 * 
100 	 * @param outputDir
101 	 * @param rootDirectory
102 	 * @throws IOException
103 	 */
104 	void copyImageResources(final File outputDir, final File rootDirectory) throws IOException {
105 		Asserts.notNull(outputDir, "outputDir");
106 		Asserts.notNull(rootDirectory, "files");
107 
108 		copyImageResources(outputDir, "", rootDirectory.listFiles());
109 	}
110 
111 	/**
112 	 * Copies the image resources ({@link Constants#IMAGE_TYPES}) to the output directory prefixing the file name with the file name prefix
113 	 * argument. The file name will also be in lower case, these steps guarantee compatibility with android resource file names
114 	 * restrictions. If a sub directory is discovered in the list of files, this method will be recursively called with the file name prefix
115 	 * of the directory name postfixed with a '_'.
116 	 * 
117 	 * @param outputDir
118 	 *            The location to copy the files to.
119 	 * @param fileNamePrefix
120 	 *            The prefix to append to the output file name.
121 	 * @param files
122 	 *            The list of files to copy.
123 	 * @throws IOException
124 	 */
125 	void copyImageResources(final File outputDir, final String fileNamePrefix, final File[] files) throws IOException {
126 		Asserts.notNull(fileNamePrefix, "fileNamePrefix");
127 
128 		for (File file : files) {
129 			if (file.isFile() && isImageFile(file)) {
130 				File destination = new File(outputDir, fileNamePrefix.toLowerCase() + file.getName().toLowerCase());
131 				getLog().debug("Copying " + file + " -> " + destination);
132 				FileUtils.copyFile(file, destination);
133 			}
134 			else {
135 				copyImageResources(outputDir, file.getName() + "_", file.listFiles());
136 			}
137 		}
138 	}
139 
140 	/**
141 	 * Checks if the file ends with an extension specified in {@link Constants#IMAGE_TYPES}.
142 	 * 
143 	 * @param fileName
144 	 * @return
145 	 */
146 	boolean isImageFile(final File file) {
147 		Asserts.notNull(file, "file");
148 
149 		String fileName = file.getName().toLowerCase();
150 		for (String imageType : Constants.IMAGE_TYPES) {
151 			if (fileName.endsWith(imageType)) {
152 				return true;
153 			}
154 		}
155 		return false;
156 	}
