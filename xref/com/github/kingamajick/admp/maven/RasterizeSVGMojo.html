51  /**
52   * Rasterizes any SVGs contained <code>${svgDirectory}</code> (default: <code>'src/main/svg'</code>) to
53   * <code>'${project.build.outputDirectory}/res'</code> for any densities specified.
54   * <ul>
55   * <li>ldpi (default)</li>
56   * <li>mdpi (default)</li>
57   * <li>hdpi (default)</li>
58   * <li>xhdpi (default)</li>
59   * <li>nodpi</li>
60   * <li>tvdpi</li>
61   * </ul>
62   * 
63   * @author R King
64   * 
65   * @goal rasterize
66   */
67  public class RasterizeSVGMojo extends AbstractMojo {

74  	private TranscoderFactory transcoderFactory;

87  	File targetDir;

93  	 * <li>name : 'drawable-mdpi', scale-factor : 1.00f</li>
94  	 * <li>name : 'drawable-hdpi', scale-factor : 1.50f</li>
95  	 * <li>name : 'drawable-xhdpi', scale-factor : 2.00f</li>

118 	static File createDirectory(final File root, final String newDirName) throws MojoFailureException {
119 		File dir = new File(root, newDirName);
120 		if (!dir.exists()) {
121 			boolean success = dir.mkdirs();
122 			if (!success) {
123 				throw new MojoFailureException("Unable to make directory " + dir.getAbsolutePath());
124 			}
125 		}

129 	public RasterizeSVGMojo() {
130 		this.userAgent = new UserAgentAdapter();
131 		this.loader = new DocumentLoader(this.userAgent);
132 		this.context = new BridgeContext(this.userAgent, this.loader);
133 		this.context.setDynamicState(BridgeContext.DYNAMIC);
134 		this.builder = new GVTBuilder();
135 		this.svgDocFactory = new SAXSVGDocumentFactory(XMLResourceDescriptor.getXMLParserClassName());
136 		this.transcoderFactory = new TranscoderFactory();
137 	}

139 	/*
140 	 * (non-Javadoc)
141 	 * 
142 	 * @see org.apache.maven.plugin.Mojo#execute()
143 	 */
144 	public void execute() throws MojoExecutionException, MojoFailureException {
145 		if (this.densities.size() == 0) {
146 			Density.defaults(this.densities);
147 		}
148 		ImageTranscoder transcoder = null;
149 		try {
150 			transcoder = this.transcoderFactory.create(this.rasterizedType);
151 		}
152 		catch (TranscoderFactoryException e) {
153 			throw new MojoExecutionException("Unable to create transcoder", e);
154 		}
155 		Map<String, File> svgsToProcess = getSVGsToProcess(this.svgDirectory);
156 		for (Entry<String, File> svgToProcess : svgsToProcess.entrySet()) {
157 			try {
158 				String inputURI = svgToProcess.getValue().toURI().toString();
159 				Document svgDoc = this.svgDocFactory.createDocument(inputURI);
160 				GraphicsNode rootGN = this.builder.build(this.context, svgDoc);
161 				Rectangle2D bounds = rootGN.getBounds();
162 
163 				for (Density density : this.densities) {
164 					getLog().debug(
165 							"Rasterizing " + svgToProcess.getValue() + " -> " + density.getName() + "/" + svgToProcess.getKey() + "." + this.rasterizedType + " [" + density.getScaleFactor() + "]");
166 
167 					File outputDir = createDirectory(this.targetDir, density.getName());
168 					File outputFile = new File(outputDir, svgToProcess.getKey() + "." + this.rasterizedType);
169 					outputFile.createNewFile();
170 					OutputStream os = new FileOutputStream(outputFile);
171 
172 					TranscoderInput input = new TranscoderInput(svgDoc);
173 					TranscoderOutput output = new TranscoderOutput(os);
174 
175 					transcoder.addTranscodingHint(ImageTranscoder.KEY_WIDTH, new Float(Math.ceil(density.getScaleFactor() * bounds.getWidth())));
176 					transcoder.transcode(input, output);
177 				}
178 
179 			}
180 			catch (IOException e) {
181 				throw new MojoFailureException("Unable to rasterize " + svgToProcess.getValue().getAbsolutePath(), e);
182 			}
183 
184 			catch (TranscoderException e) {
185 				throw new MojoFailureException("Unable to rasterize " + svgToProcess.getValue().getAbsolutePath(), e);
186 			}
187 		}
188 	}

190 	/**
191 	 * Returns a map of all SVGs contain in the rootDirectory and its sub directories, the map is keyed by the output name for that file
192 	 * when rasterized and contains the input file as the value. The output names will be in lower case, and in the case where the file is
193 	 * contained in a sub directory, the output file name will be prefixed with the directory path separated by '_', i.e
194 	 * ${subDir1}_${subDir2}_${imageFileName}.
195 	 * 
196 	 * @param rootDirectory
197 	 * @return
198 	 */
199 	Map<String, File> getSVGsToProcess(final File rootDirectory) {
200 		Asserts.notNull(rootDirectory, "rootDirectory");
201 		Map<String, File> fileMappings = new HashMap<String, File>();
202 		getSVGsToProcess(fileMappings, rootDirectory.listFiles(), "");
203 		return fileMappings;
204 	}

206 	/**
207 	 * Processes the file list for potential SVG files. If a file is found it is added to the map with a key of the file name prefixed with
208 	 * the file name prefix. If a potential SVG file is found to be a directory, it will call this method recursively to process the
209 	 * potential SVGs in this folder, with the fileNamePrefix appended by the directory name and a '_'
210 	 * 
211 	 * @param fileMappings
212 	 * @param potentialSVGs
213 	 * @param fileNamePrefix
214 	 */
215 	void getSVGsToProcess(final Map<String, File> fileMappings, final File[] potentialSVGs, final String fileNamePrefix) {
216 		for (File potentialSVG : potentialSVGs) {
217 			String potentialSVGName = fileNamePrefix + potentialSVG.getName();
218 			if (potentialSVG.isFile() && potentialSVG.getName().toLowerCase().endsWith(Constants.SVG_FILE_TYPE)) {
219 				// Output file name sans the extension
220 				String outputFileName = potentialSVGName.substring(0, potentialSVGName.length() - Constants.SVG_FILE_TYPE.length());
221 				getLog().debug("Mapping " + potentialSVG + " to " + outputFileName);
222 				fileMappings.put(outputFileName, potentialSVG);
223 			}
224 			else if (potentialSVG.isDirectory()) {
225 				getSVGsToProcess(fileMappings, potentialSVG.listFiles(), potentialSVGName + "_");
226 			}
227 			else {
228 				getLog().debug("Ignoring resource " + potentialSVG);
229 			}
230 		}
231 	}
