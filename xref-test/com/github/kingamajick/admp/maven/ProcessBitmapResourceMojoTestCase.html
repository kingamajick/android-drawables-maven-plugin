44  @RunWith(PowerMockRunner.class)

46  public class ProcessBitmapResourceMojoTestCase {

48  	private final static File TEST_DIR = new File("target/test-classes/process-bitmap-resources/");
49  	private final static String IMAGE_FILE_NAME = "ic_launcher.png";
50  	private Log mockLogger = createNiceMock(Log.class);
51  
52  	private final static class FileExpectationHolder {
53  		File inputFile;
54  		String inputFileChecksum;
55  		File expectedFile;
56  	}
57  
58  	/**
59  	 * Helper method to create a {@link FileExpectationHolder} with the given input and expected file with the holders checksum generated.
60  	 * 
61  	 * @param inputFile
62  	 * @param expectedFile
63  	 * @return
64  	 * @throws MojoFailureException
65  	 */
66  	private final static FileExpectationHolder createExpectation(final File inputFile, final File expectedFile) throws MojoFailureException {
67  		FileExpectationHolder holder = new FileExpectationHolder();
68  		holder.inputFile = inputFile;
69  		holder.inputFileChecksum = TestUtils.getChecksum(inputFile);
70  		holder.expectedFile = expectedFile;
71  		return holder;
72  	}

74  	@Test
75  	public void successfulCopy() throws Exception {
76  		// Setup
77  		File testBaseDir = new File(TEST_DIR, "baseCase");
78  		File resouceDir = new File(testBaseDir, "src/main/resources");
79  		File targetDir = new File(testBaseDir, "target/classes/res");
80  
81  		Resource mockResource = createMock(Resource.class);
82  		List<Resource> resources = new ArrayList<Resource>();
83  		resources.add(mockResource);
84  
85  		ProcessBitmapResourcesMojo mojo = new ProcessBitmapResourcesMojo();
86  		Reflection.field("resources").ofType(List.class).in(mojo).set(resources);
87  		Reflection.field("targetDir").ofType(File.class).in(mojo).set(targetDir);
88  		mojo.setLog(this.mockLogger);
89  
90  		// @formatter:off
91  		FileExpectationHolder[] expectedFiles = {
92  			createExpectation(new File(resouceDir, "ldpi/"  + IMAGE_FILE_NAME), new File(targetDir, "drawable-ldpi/"  + IMAGE_FILE_NAME)),
93  			createExpectation(new File(resouceDir, "mdpi/"  + IMAGE_FILE_NAME), new File(targetDir, "drawable-mdpi/"  + IMAGE_FILE_NAME)),
94  			createExpectation(new File(resouceDir, "hdpi/"  + IMAGE_FILE_NAME), new File(targetDir, "drawable-hdpi/"  + IMAGE_FILE_NAME)),
95  			createExpectation(new File(resouceDir, "xhdpi/" + IMAGE_FILE_NAME), new File(targetDir, "drawable-xhdpi/" + IMAGE_FILE_NAME)),
96  			createExpectation(new File(resouceDir, "nodpi/" + IMAGE_FILE_NAME), new File(targetDir, "drawable-nodpi/" + IMAGE_FILE_NAME)),
97  			createExpectation(new File(resouceDir, "tvdpi/" + IMAGE_FILE_NAME), new File(targetDir, "drawable-tvdpi/" + IMAGE_FILE_NAME)),
98  				
99  		};
100 		// @formatter:on 
101 
102 		// Expectations
103 		expect(mockResource.getDirectory()).andReturn(resouceDir.getAbsolutePath());
104 
105 		replay(mockResource);
106 
107 		// Execute
108 		mojo.execute();
109 
110 		// Assert
111 		for (FileExpectationHolder holder : expectedFiles) {
112 			assertTrue("Expected file " + holder.expectedFile.getAbsolutePath() + " does not exist.", holder.expectedFile.exists());
113 			assertEquals("Checksums don't match for files " + holder.inputFile + " and " + holder.expectedFile, holder.inputFileChecksum, TestUtils.getChecksum(holder.expectedFile));
114 
115 		}
116 		verify(mockResource);
117 	}
