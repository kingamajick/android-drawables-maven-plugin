46  	private final static String PNG_TYPE = "png";
47  	private final static File TEST_DIR = new File("target/test-classes/rasterize-svgs/");
48  	private Log mockLogger = createNiceMock(Log.class);

50  	private final static class PNGExpectationHolder {
51  		File expectedFile;
52  		int expectedWidth;
53  		int expectedHeight;
54  	}
55  
56  	/**
57  	 * Helper method to create a {@link PNGExpectationHolder} with the given expected file, width and height.
58  	 * 
59  	 * @param expectedFile
60  	 * @param expectedWidth
61  	 * @param expectedHeight
62  	 * @return
63  	 */
64  	private final static PNGExpectationHolder createExpectation(final File expectedFile, final int expectedWidth, final int expectedHeight) {
65  		PNGExpectationHolder holder = new PNGExpectationHolder();
66  		holder.expectedFile = expectedFile;
67  		holder.expectedWidth = expectedWidth;
68  		holder.expectedHeight = expectedHeight;
69  		return holder;
70  	}

72  	@Test
73  	public void successfulRaterizationDefaultDensities() throws Exception {
74  		// Setup
75  		File testBaseDir = new File(TEST_DIR, "baseCase");
76  		File svgDir = new File(testBaseDir, "src/main/svg");
77  		File targetDir = new File(testBaseDir, "target/classes/res");
78  		File ldpiDir = new File(targetDir, "drawable-ldpi");
79  		File mdpiDir = new File(targetDir, "drawable-mdpi");
80  		File hdpiDir = new File(targetDir, "drawable-hdpi");
81  		File xhdpiDir = new File(targetDir, "drawable-xhdpi");
82  		String subDirFile = "icon_main.png";
83  		String rootFile = "image.png";
84  
85  		RasterizeSVGMojo mojo = new RasterizeSVGMojo();
86  		mojo.setLog(this.mockLogger);
87  
88  		Reflection.field("svgDirectory").ofType(File.class).in(mojo).set(svgDir);
89  		Reflection.field("targetDir").ofType(File.class).in(mojo).set(targetDir);
90  		Reflection.field("densities").ofType(List.class).in(mojo).set(new ArrayList<Density>());
91  		Reflection.field("rasterizedType").ofType(String.class).in(mojo).set(PNG_TYPE);
92  
93  		// @formatter:off
94  		PNGExpectationHolder[] expectedPNGs = {
95  				createExpectation(new File(ldpiDir,  subDirFile), 36, 36),
96  				createExpectation(new File(mdpiDir,  subDirFile), 48, 48),
97  				createExpectation(new File(hdpiDir,  subDirFile), 72, 72),
98  				createExpectation(new File(xhdpiDir, subDirFile), 96, 96),
99  				
100 				createExpectation(new File(ldpiDir,  rootFile),  150, 75),
101 				createExpectation(new File(mdpiDir,  rootFile),  200, 100),
102 				createExpectation(new File(hdpiDir,  rootFile),  300, 150),
103 				createExpectation(new File(xhdpiDir, rootFile),  400, 200),
104 		};
105 		// @formatter:on
106 
107 		// Execute
108 		mojo.execute();
109 
110 		// Assert
111 		for (PNGExpectationHolder holder : expectedPNGs) {
112 			assertTrue("Expected file " + holder.expectedFile.getAbsolutePath() + " does not exist.", holder.expectedFile.exists());
113 			BufferedImage img = ImageIO.read(holder.expectedFile);
114 			assertEquals("Width of generated PNG is " + img.getWidth() + "px, expected " + holder.expectedWidth + "px", holder.expectedWidth, img.getWidth());
115 			assertEquals("Height of generated PNG is " + img.getHeight() + "px, expected " + holder.expectedHeight + "px", holder.expectedHeight, img.getHeight());
116 		}
117 	}
