50  	@Test
51  	public void successfulUnpack() throws Exception {
52  		// Setup
53  		RepositorySystem mockRepositorySystem = createMock(RepositorySystem.class);
54  		ArtifactRepository mockLocalRepository = createMock(ArtifactRepository.class);
55  		List<ArtifactRepository> remoteRepositories = Collections.emptyList();
56  
57  		File testBaseDir = new File(TEST_DIR, "baseCase");
58  		File unpackLocation = new File(testBaseDir, "target/unpack");
59  		File drawableArtifactZip = new File(testBaseDir, "drawable-artifact.zip");
60  
61  		String groupId = "org.test";
62  		String artifactId = "drawable-artifact";
63  		String version = "0.0.0";
64  		List<DrawableArtifact> drawableArtifacts = new ArrayList<DrawableArtifact>();
65  		DrawableArtifact drawableArtifact = new DrawableArtifact(groupId, artifactId, version);
66  		drawableArtifacts.add(drawableArtifact);
67  
68  		Artifact mockArtifact = createMock(Artifact.class);
69  
70  		UnpackBitmapResourcesMojo mojo = new UnpackBitmapResourcesMojo();
71  		Reflection.field("repositorySystem").ofType(RepositorySystem.class).in(mojo).set(mockRepositorySystem);
72  		Reflection.field("localRepository").ofType(ArtifactRepository.class).in(mojo).set(mockLocalRepository);
73  		Reflection.field("remoteRepositories").ofType(new TypeRef<List<ArtifactRepository>>() {}).in(mojo).set(remoteRepositories);
74  		Reflection.field("unpackLocation").ofType(File.class).in(mojo).set(unpackLocation);
75  		Reflection.field("drawableArtifacts").ofType(new TypeRef<List<DrawableArtifact>>() {}).in(mojo).set(drawableArtifacts);
76  		mojo.setLog(this.mockLogger);
77  
78  		// Expectations
79  		mockRepositorySystem.createArtifact(groupId, artifactId, version, "android-drawables");
80  		expectLastCall().andReturn(mockArtifact);
81  		mockRepositorySystem.resolve(anyObject(ArtifactResolutionRequest.class));
82  		expectLastCall().andReturn(createNiceMock(ArtifactResolutionResult.class));
83  		mockArtifact.isResolved();
84  		expectLastCall().andReturn(true);
85  		mockArtifact.getFile();
86  		expectLastCall().andReturn(drawableArtifactZip).atLeastOnce();
87  
88  		replay(mockRepositorySystem, mockLocalRepository, mockArtifact);
89  
90  		// Execute
91  		mojo.execute();
92  
93  		// Assert
94  		File unpackedImageFile = new File(testBaseDir, "target/unpack/res/drawable-nodpi/image.png");
95  		assertTrue(unpackedImageFile.exists());
96  		assertEquals(TEST_PNG_MD5CHSUM, TestUtils.getChecksum(unpackedImageFile));
97  	}
